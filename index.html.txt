// ============================================================
// BACKGROUND SYNC

function enqueuSync(fn, label) {
  syncQueue.push({ fn, label });
  runSyncQueue();
}

async function runSyncQueue() {
  if(syncRunning || syncQueue.length === 0) return;
  syncRunning = true;
  document.getElementById('syncIndicator').classList.add('active');
  while(syncQueue.length > 0) {
    const job = syncQueue.shift();
    document.getElementById('syncQueueText').textContent = job.label || '同步中...';
    document.getElementById('syncQueue').classList.add('show');
    try { await job.fn(); } catch(e) { console.warn('Background sync failed:', e); }
  }
  syncRunning = false;
  document.getElementById('syncIndicator').classList.remove('active');
  document.getElementById('syncQueue').classList.remove('show');
}

function renderLedgerTabs() {
  const dates = Object.keys(ledger).sort().reverse();
  if(dates.length === 0) { ensureTodayLedger(); return renderLedgerTabs(); }
  if(!currentLedgerDate || !ledger[currentLedgerDate]) currentLedgerDate = dates[0];

  document.getElementById('ledgerTabs').innerHTML = dates.map(d => {
    const count = (ledger[d]||[]).length;
    const label = formatDateLabel(d);
    return `<button class="ledger-tab-btn ${d===currentLedgerDate?'active':''}" onclick="switchLedger('${d}')">
      ${label} <span class="tab-count">${count}</span>
    </button>`;
  }).join('');

  // 更新下拉菜單
  const sel = document.getElementById('ledgerSelector');
  if(sel) {
    sel.innerHTML = '<option value="">-- 選擇帳務本 --' + 
      dates.map(d => `<option value="${d}" ${d===currentLedgerDate?'selected':''}>${formatDateLabel(d)}</option>`).join('');
    sel.value = currentLedgerDate;
  }

  const cl = document.getElementById('chartLedger');
  if(cl) {
    cl.innerHTML = '<option value="all">全部帳本</option>' +
      dates.map(d => `<option value="${d}">${formatDateLabel(d)}</option>`).join('');
  }
}

function formatDateLabel(d) {
  const dt = new Date(d + 'T00:00:00');
  return dt.toLocaleDateString('zh-TW', {month:'2-digit',day:'2-digit'}) + ' 帳本';
}

}

function getFilteredStocks() {
  let data = STOCKS_RAW;
  if(currentCat !== 'all') data = data.filter(s => s['分類'] === currentCat);
  const q = (document.getElementById('searchInput').value || '').toLowerCase();
  if(q) data = data.filter(s => s['代號'].includes(q) || s['名稱'].toLowerCase().includes(q));
  const mode = document.getElementById('modeSelect').value;
  if(mode === 'top20') data = [...data].sort((a,b)=>b['貝氏評分']-a['貝氏評分']).slice(0,20);
  else if(mode === 'manual') data = data.filter(s => manualSelected.has(s['代號']));
  return data;
}

function calcShares(price) {
  if(!price || price <= 0) return 0;
  return Math.floor(1000 / price);
}

function sortBy(col) {
  if(sortCol===col) sortDir*=-1; else { sortCol=col; sortDir=-1; }
  renderPerf();
}

function enrichStock(s) {
  const price = livePrices[s['代號']] || s['現價'];
  const shares = calcShares(s['現價']);
  const invested = shares * s['現價'];
  const curVal = shares * price;
  const pnl = curVal - invested;
  const pct = invested > 0 ? pnl/invested*100 : 0;
  return {...s, _live:price, _shares:shares, _pnl:pnl, _pct:pct, _invested:invested};
}

function renderPerf() {
  let data = getFilteredStocks().map(enrichStock);
  const colMap = {'pnl':'_pnl','pct':'_pct','現價':'_live'};
  const sc = colMap[sortCol] || sortCol;
  data.sort((a,b) => {
    let av=a[sc],bv=b[sc];
    if(typeof av==='string') av=av.toLowerCase();
    if(typeof bv==='string') bv=bv.toLowerCase();
    if(av==null) av=-Infinity; if(bv==null) bv=-Infinity;
    return sortDir*(av>bv?1:av<bv?-1:0);
  });
  document.getElementById('countBadge').textContent = data.length + ' 檔';

  document.getElementById('perfBody').innerHTML = data.map(s => {
    const chg = parseFloat(s['漲幅']) || null;
    const chgCls = chg===null?'neu':chg>0?'pos':chg<0?'neg':'neu';
    const chgStr = chg===null?'--':(chg>0?'+':'')+chg+'%';
    const pnlCls = s._pnl>0?'pos':s._pnl<0?'neg':'neu';
    const cc = CAT_COLORS[s['分類']]||'#888';
    const cl = CAT_LABELS[s['分類']]||s['分類'];
    const pnlStr = s._shares>0?(s._pnl>=0?'+':'')+s._pnl.toFixed(0):'--';
    const pctStr = s._shares>0?(s._pct>=0?'+':'')+s._pct.toFixed(1)+'%':'--';
    return `<tr>
      <td class="code">${s['代號']}</td>
      <td class="name">${s['名稱']}</td>
      <td><span class="cat-tag" style="background:${cc}22;color:${cc}">${cl}</span></td>
      <td class="score" style="color:var(--gold)">${s['貝氏評分'].toFixed(2)}</td>
      <td class="price-cell">${s['現價'].toFixed(2)}</td>
      <td class="change-cell ${chgCls}">${chgStr}</td>
      <td class="price-cell">${s._shares>0?s._shares+' 股':'--'}</td>
      <td class="pnl-cell ${pnlCls}">${pnlStr}</td>
      <td class="pct-cell ${pnlCls}">${pctStr}</td>
      <td><button class="btn ghost" style="padding:3px 9px;font-size:11px;" onclick="addOneLedger('${s['代號']}','${s['名稱']}','${s['分類']}',${s['現價']},${s['貝氏評分']})">＋帳本</button></td>
    </tr>`;
  }).join('');
  renderSummaryCards(data);
}

function renderSummaryCards(data) {
  if(!data) data = getFilteredStocks().map(enrichStock);
  const posCount = data.filter(s=>s._pnl>0).length;
  const negCount = data.filter(s=>s._pnl<0).length;
  const totalPnl = data.reduce((a,s)=>a+(s._pnl||0),0);
  const avgScore = data.length?(data.reduce((a,s)=>a+s['貝氏評分'],0)/data.length).toFixed(2):'--';
  const pnlCls = totalPnl>=0?'pos':'neg';
  const pnlStr = (totalPnl>=0?'+':'')+totalPnl.toFixed(0);
  document.getElementById('summaryCards').innerHTML = `
    <div class="stat-card"><div class="label">篩選檔數</div><div class="value neu">${data.length}</div></div>
    <div class="stat-card"><div class="label">上漲</div><div class="value pos">${posCount}</div></div>
    <div class="stat-card"><div class="label">下跌</div><div class="value neg">${negCount}</div></div>
    <div class="stat-card"><div class="label">預估總損益</div><div class="value ${pnlCls}">${pnlStr}</div><div class="sub">NTD (各1000元)</div></div>
    <div class="stat-card"><div class="label">平均貝氏評分</div><div class="value" style="color:var(--gold)">${avgScore}</div></div>
  `;
}

function addOneLedger(code, name, cat, price, score) {
  const targetDate = document.getElementById('saveLedgerSelect').value;
  const today = new Date().toISOString().split('T')[0];
  const shares = calcShares(price);
  if(shares===0) { showToast('價格過高，1000元買不到1股','fail'); return; }
  if(!ledger[targetDate]) ledger[targetDate] = [];
  if(ledger[targetDate].find(r => r.code===code && r.date===today)) { showToast(name+' 今日已在帳本中','fail'); return; }
  ledger[targetDate].push({ date:today, code, name, cat, buyPrice:price, shares, score });
  renderLedgerTabs();
  showToast('已加入 '+name+' 到'+targetDate+'帳本（需點「☁️ 同步GAS」以保存）','ok');
}

async function saveTodayToGAS() {
  const targetDate = document.getElementById('saveLedgerSelect').value;
  const data = getFilteredStocks();
  const today = new Date().toISOString().split('T')[0];
  let added = 0;
  if(!ledger[targetDate]) ledger[targetDate] = [];
  data.forEach(s => {
    const shares = calcShares(s['現價']);
    if(shares>0 && !ledger[targetDate].find(r=>r.code===s['代號']&&r.date===today)) {
      ledger[targetDate].push({ date:today, code:s['代號'], name:s['名稱'], cat:s['分類'], buyPrice:s['現價'], shares, score:s['貝氏評分'] });
      added++;
    }
  });
  renderLedgerTabs();
  showToast('已加入 '+added+' 檔到'+targetDate+'帳本（需點「☁️ 同步GAS」以保存）','ok');
}

function getDateRange(period) {
  const now = new Date();
  if(period==='all') return null;
  if(period==='day') {
    const d = now.toISOString().split('T')[0];
    return { from:d, to:d };
  }
  if(period==='month') {
    const y = now.getFullYear(), m = String(now.getMonth()+1).padStart(2,'0');
    return { from:`${y}-${m}-01`, to:`${y}-${m}-31` };
  }
  if(period==='year') {
    return { from:`${now.getFullYear()}-01-01`, to:`${now.getFullYear()}-12-31` };
  }
  return null;
}

function renderLedger() {
  if(!currentLedgerDate || !ledger[currentLedgerDate]) { ensureTodayLedger(); }

  const banner = document.getElementById('ledgerPriceBanner');
  const isStale = !livePriceTimestamp || (Date.now() - livePriceTimestamp > 3600000);
  banner.style.display = isStale ? 'block' : 'none';

  let data = (ledger[currentLedgerDate] || []).map((r,idx) => ({...r, _idx:idx}));

  const range = getDateRange(currentPeriod);
  if(range) data = data.filter(r => r.date >= range.from && r.date <= range.to);

  const dateF = document.getElementById('ledgerDateFilter').value;
  const catF = document.getElementById('ledgerCatFilter').value;
  const q = (document.getElementById('ledgerSearch').value||'').toLowerCase();
  const pnlF = document.getElementById('ledgerPnlFilter').value;

  if(dateF) data = data.filter(r => r.date===dateF);
  if(catF) data = data.filter(r => r.cat===catF);
  if(q) data = data.filter(r => r.code.includes(q)||r.name.toLowerCase().includes(q));

  if(pnlF) {
    data = data.filter(r => {
      const lp = livePrices[r.code]||r.buyPrice;
      const pct = r.buyPrice>0?(lp-r.buyPrice)/r.buyPrice*100:0;
      if(pnlF==='profit') return pct>0;
      if(pnlF==='loss') return pct<0;
      if(pnlF==='small') return Math.abs(pct)<=5;
      return true;
    });
  }

  const allDates = [...new Set((ledger[currentLedgerDate]||[]).map(r=>r.date))].sort().reverse();
  const selDate = document.getElementById('ledgerDateFilter').value;
  document.getElementById('ledgerDateFilter').innerHTML = '<option value="">全部日期</option>' +
    allDates.map(d=>`<option value="${d}" ${d===selDate?'selected':''}>${d}</option>`).join('');

  document.getElementById('ledgerEmpty').style.display = data.length?'none':'block';
  document.getElementById('ledgerTable').style.display = data.length?'table':'none';

  document.getElementById('ledgerBody').innerHTML = data.map(r => {
    const lp = livePrices[r.code]||r.buyPrice;
    const isLive = !!livePrices[r.code];
    const invested = r.shares*r.buyPrice;
    const curVal = r.shares*lp;
    const pnl = curVal-invested;
    const pct = invested>0?pnl/invested*100:0;
    const pnlCls = pnl>=0?'pos':'neg';
    const cc = CAT_COLORS[r.cat]||'#888';
    const cl = CAT_LABELS[r.cat]||r.cat;
    const liveBadge = isLive
      ? `<span class="live-badge">即時</span>`
      : `<span class="live-badge stale">估算</span>`;
    return `<tr>
      <td class="code" style="font-size:11px;">${r.date}</td>
      <td class="code">${r.code}</td>
      <td class="name">${r.name}</td>
      <td><span class="cat-tag" style="background:${cc}22;color:${cc}">${cl}</span></td>
      <td class="price-cell">${r.buyPrice.toFixed(2)}</td>
      <td class="price-cell">${r.shares}</td>
      <td class="price-cell">${invested.toFixed(0)}</td>
      <td class="price-cell">${lp.toFixed(2)}${liveBadge}</td>
      <td class="price-cell">${curVal.toFixed(0)}</td>
      <td class="pnl-cell ${pnlCls}">${pnl>=0?'+':''}${pnl.toFixed(0)}</td>
      <td class="pct-cell ${pnlCls}">${pct>=0?'+':''}${pct.toFixed(1)}%</td>
      <td><button class="btn red" style="padding:3px 8px;font-size:11px;" onclick="deleteLedger('${currentLedgerDate}',${r._idx})">刪除</button></td>
    </tr>`;
  }).join('');

  const ti = data.reduce((a,r)=>a+r.shares*r.buyPrice,0);
  const tv = data.reduce((a,r)=>a+r.shares*(livePrices[r.code]||r.buyPrice),0);
  const tp = tv-ti;
  const tpct = ti>0?tp/ti*100:0;
  const pc = tp>=0?'pos':'neg';
  const winRate = data.length>0?Math.round(data.filter(r=>{
    const lp=livePrices[r.code]||r.buyPrice; return lp>r.buyPrice;
  }).length/data.length*100):0;

  document.getElementById('ledgerSummary').innerHTML = `
    <div class="stat-card"><div class="label">帳務筆數</div><div class="value neu">${data.length}</div></div>
    <div class="stat-card"><div class="label">總投入</div><div class="value neu">${ti.toFixed(0)}</div><div class="sub">NTD</div></div>
    <div class="stat-card"><div class="label">總市值</div><div class="value neu">${tv.toFixed(0)}</div><div class="sub">NTD</div></div>
    <div class="stat-card"><div class="label">總損益</div><div class="value ${pc}">${tp>=0?'+':''}${tp.toFixed(0)}</div><div class="sub">NTD</div></div>
    <div class="stat-card"><div class="label">損益率</div><div class="value ${pc}">${tpct>=0?'+':''}${tpct.toFixed(2)}%</div></div>
    <div class="stat-card"><div class="label">勝率</div><div class="value ${winRate>=50?'pos':'neg'}">${winRate}%</div></div>
  `;
}

function applyManualPrices() {
  const input = document.getElementById('manualPriceInput').value.trim();
  const status = document.getElementById('manualPriceStatus');
  
  if(!input) {
    status.textContent = '❌ 請輸入價格數據';
    status.style.color = 'var(--red)';
    return;
  }
  
  const lines = input.split('\n').filter(l => l.trim());
  let success = 0, failed = 0;
  const failedLines = [];
  
  lines.forEach(line => {
    const parts = line.split('=').map(s => s.trim());
    if(parts.length !== 2) {
      failed++;
      failedLines.push(line);
      return;
    }
    
    const [code, priceStr] = parts;
    const price = parseFloat(priceStr);
    
    if(!code || isNaN(price) || price <= 0) {
      failed++;
      failedLines.push(line);
      return;
    }
    
    livePrices[code] = price;
    console.log(`✓ 手動輸入: ${code} = ${price}`);
    success++;
  });
  
  if(success > 0) {
    status.textContent = `✓ 成功套用 ${success} 筆現價`;
    status.style.color = 'var(--green)';
    livePriceTimestamp = Date.now();
    renderPerf();
    renderLedger();
    setTimeout(() => {
      document.getElementById('manualPriceModal').classList.remove('open');
      showToast(`已手動輸入 ${success} 筆現價`, 'ok');
    }, 800);
  } else {
    status.textContent = `❌ 全部輸入有誤`;
    status.style.color = 'var(--red)';
  }
  
  if(failed > 0) {
    console.warn(`有 ${failed} 行格式錯誤:`, failedLines);
  }
}

// LIVE PRICE UPDATE - 批量並列抓取 (BATCH PARALLEL - 方案A)
// 原本：逐檔串列 + 120ms 延遲 → 50檔約 6秒以上
// 優化後：5檔一批並列 + 80ms 批次間延遲 → 50檔約 1-2秒

function showProgress(current, total) {
  const pct = Math.round(current / total * 100);
  const wrap = document.getElementById('progressWrap');
  const bar = document.getElementById('progressBar');
  if(wrap) wrap.style.display = 'block';
  if(bar) bar.style.width = pct + '%';
  document.getElementById('loadingText').textContent =
    `更新現價中... ${current}/${total} (${pct}%)`;
}

function hideProgress() {
  const wrap = document.getElementById('progressWrap');
  if(wrap) wrap.style.display = 'none';
  const bar = document.getElementById('progressBar');
  if(bar) bar.style.width = '0%';
}

async function fetchWithTimeout(url, timeoutMs = 8000) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);
  try {
    const res = await fetch(url, {signal: controller.signal, headers: {'Accept': 'application/json'}});
    clearTimeout(timeoutId);
    return res;
  } catch(e) {
    clearTimeout(timeoutId);
    throw e;
  }
}

async function fetchOneLivePrice(code) {
  console.log(`🔍 開始查詢: ${code}`);
  
  // 嘗試查詢 TWSE/OTC API
  for(let attempt = 0; attempt < 2; attempt++) {
    const isOTC = attempt === 1;
    const exchange = isOTC ? 'otc' : 'tse';
    const url = `https://mis.twse.com.tw/stock/api/getStockInfo.jsp?json=1&delay=5&ex_ch=${exchange}_${code}.tw`;
    
    try {
      console.debug(`  [${attempt + 1}/2] 嘗試 ${exchange.toUpperCase()} API...`);
      const res = await fetchWithTimeout(url, 10000);
      
      if(!res.ok) {
        console.debug(`  ✗ HTTP ${res.status}`);
        continue;
      }
      
      const d = await res.json();
      
      if(!d.msgArray || d.msgArray.length === 0) {
        console.debug(`  ✗ 無資料`);
        continue;
      }
      
      const z = d.msgArray[0].z;
      if(z && z !== '-' && z !== '') {
        const price = parseFloat(z);
        if(price > 0) {
          console.log(`  ✓ ${exchange.toUpperCase()} 查詢成功: ${code} = ${price}`);
          return price;
        }
      }
      console.debug(`  ✗ 價格無效: ${z}`);
      
    } catch(e) {
      console.debug(`  ✗ ${exchange.toUpperCase()} 錯誤: ${e.message}`);
    }
  }
  
  console.warn(`❌ 無法查詢 ${code} - TWSE API 失敗，使用本地緩存價格`);
  return null;
}

async function fetchLivePrices() {
  const visible = getFilteredStocks().slice(0, 60);
  const codes = visible.map(s => s['代號']);
  if(codes.length === 0) { showToast('沒有股票可更新', 'fail'); return; }

  showLoading(`更新現價中... 0/${codes.length}`);
  showProgress(0, codes.length);

  const BATCH = 2;       // 每批只查 2 檔
  const DELAY_MS = 500;   // 批次間隔增加到 500ms
  let updated = 0;
  let failed = [];
  let done = 0;

  console.log(`=== 開始查詢 ${codes.length} 檔股票現價 ===`);

  for(let i = 0; i < codes.length; i += BATCH) {
    const batch = codes.slice(i, i + BATCH);

    const results = await Promise.allSettled(
      batch.map(async code => {
        try {
          const price = await fetchOneLivePrice(code);
          if(price !== null) { 
            livePrices[code] = price; 
            updated++; 
          } else {
            failed.push(code);
          }
        } catch(e) {
          failed.push(code);
          console.error(`查詢異常 ${code}:`, e.message);
        }
      })
    );

    done = Math.min(i + BATCH, codes.length);
    showProgress(done, codes.length);
    document.getElementById('loadingText').textContent = `更新現價中... ${done}/${codes.length} (成功: ${updated})`;

    if(i + BATCH < codes.length) {
      await new Promise(r => setTimeout(r, DELAY_MS));
    }
  }

  livePriceTimestamp = Date.now();
  hideProgress();
  hideLoading();
  
  console.log(`=== 查詢完成: 成功 ${updated}/${codes.length} 檔 ===`);
  if(failed.length > 0) {
    console.warn(`失敗股票 (${failed.length}): `, failed.join(', '));
  }
  
  let msg;
  if(updated === codes.length) {
    msg = `✓ 全部 ${updated} 檔現價更新成功`;
  } else if(updated > 0) {
    msg = `⚠️ 部分成功: ${updated}/${codes.length} 檔更新 (${failed.length}檔失敗)`;
  } else {
    msg = `✗ 現價查詢全部失敗，請檢查網路連線或手動輸入`;
  }
  
  showToast(msg, updated > 0 ? 'ok' : 'fail');
  renderPerf();
  renderLedger();
}

async function fetchLivePricesForLedger() {
  if(!currentLedgerDate || !ledger[currentLedgerDate] || ledger[currentLedgerDate].length === 0) {
    showToast('帳本是空的', 'fail'); return;
  }

  const codes = [...new Set(ledger[currentLedgerDate].map(r => r.code))];
  showLoading(`更新 ${codes.length} 檔現價...`);
  showProgress(0, codes.length);

  const BATCH = 3;
  const DELAY_MS = 200;
  let updated = 0;
  let failed = [];
  let done = 0;

  console.log(`開始查詢帳本 ${codes.length} 檔股票現價...`);

  for(let i = 0; i < codes.length; i += BATCH) {
    const batch = codes.slice(i, i + BATCH);

    await Promise.allSettled(
      batch.map(async code => {
        try {
          const price = await fetchOneLivePrice(code);
          if(price !== null) { 
            livePrices[code] = price; 
            updated++; 
          } else {
            failed.push(code);
          }
        } catch(e) {
          failed.push(code);
        }
      })
    );

    done = Math.min(i + BATCH, codes.length);
    showProgress(done, codes.length);
    document.getElementById('loadingText').textContent = `更新現價中... ${done}/${codes.length} (成功: ${updated})`;

    if(i + BATCH < codes.length) {
      await new Promise(r => setTimeout(r, DELAY_MS));
    }
  }

  livePriceTimestamp = Date.now();
  hideProgress();
  hideLoading();
  
  console.log(`帳本查詢完成: 成功${updated}檔，失敗${failed.length}檔`);
  const msg = updated > 0 ? `已更新 ${updated}/${codes.length} 檔現價` : '現價查詢失敗，請檢查網路';
  showToast(msg, updated > 0 ? 'ok' : 'fail');
  renderLedger();
}

function exportLedger(type) {
  document.getElementById('ledgerExportMenu').classList.remove('open');
  const now = new Date();
  const today = now.toISOString().split('T')[0];

  const makeRows = (records) => records.map(r => {
    const lp = livePrices[r.code]||r.buyPrice;
    const invested = r.shares*r.buyPrice;
    const curVal = r.shares*lp;
    const pnl = curVal-invested;
    const pct = invested>0?pnl/invested*100:0;
    return {
      '帳本日期': r._ledgerKey||currentLedgerDate,
      '交易日期': r.date,
      '代號': r.code,
      '名稱': r.name,
      '分類': CAT_LABELS[r.cat]||r.cat,
      '貝氏評分': r.score,
      '買入價': r.buyPrice,
      '股數': r.shares,
      '投入金額': invested.toFixed(0),
      '現價': lp.toFixed(2),
      '市值': curVal.toFixed(0),
      '損益': pnl.toFixed(0),
      '損益%': pct.toFixed(2)+'%'
    };
  });

  if(type==='current-csv') {
    const rows = makeRows(ledger[currentLedgerDate]||[]);
    exportCsv(rows, `帳本_${currentLedgerDate}`);
  } else if(type==='current-xlsx') {
    const rows = makeRows(ledger[currentLedgerDate]||[]);
    exportXlsx(rows, `帳本_${currentLedgerDate}`);
  } else if(type==='period-day') {
    const allRows = [];
    Object.keys(ledger).forEach(k => {
      (ledger[k]||[]).forEach(r => {
        if(r.date===today) allRows.push({...r,_ledgerKey:k});
      });
    });
    exportXlsxMultisheet({[today]: makeRows(allRows)}, `日報表_${today}`);
  } else if(type==='period-month') {
    const y=now.getFullYear(), m=String(now.getMonth()+1).padStart(2,'0');
    const prefix = `${y}-${m}`;
    const byDate = {};
    Object.keys(ledger).forEach(k => {
      (ledger[k]||[]).forEach(r => {
        if(r.date.startsWith(prefix)) {
          if(!byDate[r.date]) byDate[r.date]=[];
          byDate[r.date].push({...r,_ledgerKey:k});
        }
      });
    });
    const sheets = {};
    Object.keys(byDate).sort().forEach(d => { sheets[d]=makeRows(byDate[d]); });
    if(Object.keys(sheets).length===0) { showToast('本月無帳務資料','fail'); return; }
    exportXlsxMultisheet(sheets, `月報表_${y}${m}`);
  } else if(type==='period-year') {
    const y=now.getFullYear();
    const byMonth = {};
    Object.keys(ledger).forEach(k => {
      (ledger[k]||[]).forEach(r => {
        if(r.date.startsWith(String(y))) {
          const mo = r.date.substring(0,7);
          if(!byMonth[mo]) byMonth[mo]=[];
          byMonth[mo].push({...r,_ledgerKey:k});
        }
      });
    });
    const sheets = {};
    Object.keys(byMonth).sort().forEach(m => { sheets[m]=makeRows(byMonth[m]); });
    if(Object.keys(sheets).length===0) { showToast('今年無帳務資料','fail'); return; }
    exportXlsxMultisheet(sheets, `年報表_${y}`);
  } else if(type==='all-xlsx') {
    const sheets = {};
    Object.keys(ledger).sort().forEach(k => {
      sheets[k] = makeRows((ledger[k]||[]).map(r=>({...r,_ledgerKey:k})));
    });
    if(Object.keys(sheets).length===0) { showToast('無帳務資料','fail'); return; }
    exportXlsxMultisheet(sheets, `全帳本_${today}`);
  } else if(type==='simple-summary') {
    // 簡易損益匯出：期間總損益和各分類損益
    const rows = [];
    const now = new Date();
    rows.push({'項目':'報告日期','值':today});
    rows.push({'項目':'統計週期','值':'全部'});
    rows.push({'項目':'','值':''});

    // 計算全部損益
    let totalInvested = 0, totalCurrentVal = 0;
    const catStats = {};
    Object.keys(ledger).forEach(ledgerKey => {
      (ledger[ledgerKey]||[]).forEach(r => {
        const lp = livePrices[r.code]||r.buyPrice;
        const invested = r.shares*r.buyPrice;
        const curVal = r.shares*lp;
        totalInvested += invested;
        totalCurrentVal += curVal;
        if(!catStats[r.cat]) catStats[r.cat] = {invested:0,current:0};
        catStats[r.cat].invested += invested;
        catStats[r.cat].current += curVal;
      });
    });
    
    const totalPnl = totalCurrentVal - totalInvested;
    const totalPct = totalInvested>0?totalPnl/totalInvested*100:0;
    rows.push({'項目':'總投入 (NTD)','值':totalInvested.toFixed(0)});
    rows.push({'項目':'總市值 (NTD)','值':totalCurrentVal.toFixed(0)});
    rows.push({'項目':'總損益 (NTD)','值':totalPnl.toFixed(0)});
    rows.push({'項目':'總損益率 (%)','值':totalPct.toFixed(2)});
    rows.push({'項目':'','值':''});
    rows.push({'項目':'分類','投入金額':'投入金額','市值':'市值','損益':'損益','損益率':'損益率(%)'});
    
    Object.entries(catStats).sort().forEach(([cat, stats]) => {
      const catPnl = stats.current - stats.invested;
      const catPct = stats.invested>0?catPnl/stats.invested*100:0;
      rows.push({
        '項目': CAT_LABELS[cat]||cat,
        '投入金額': stats.invested.toFixed(0),
        '市值': stats.current.toFixed(0),
        '損益': catPnl.toFixed(0),
        '損益率': catPct.toFixed(2)
      });
    });

    exportCsv(rows, `損益簡表_${today}`);
  }
}

function exportCsv(rows, filename) {
  if(!rows.length) { showToast('無資料可匯出','fail'); return; }
  const headers = Object.keys(rows[0]);
  const bom = '\uFEFF';
  const csv = bom + [headers.join(','), ...rows.map(r=>headers.map(h=>`"${(r[h]??'').toString().replace(/"/g,'""')}"`).join(','))].join('\n');
  const blob = new Blob([csv], {type:'text/csv;charset=utf-8'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a'); a.href=url; a.download=filename+'.csv'; a.click();
  URL.revokeObjectURL(url);
  showToast('已匯出 '+filename+'.csv','ok');
}

function exportXlsx(rows, filename) {
  if(!rows.length) { showToast('無資料可匯出','fail'); return; }
  const ws = XLSX.utils.json_to_sheet(rows);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, '資料');
  styleXlsxHeader(ws, Object.keys(rows[0]).length);
  XLSX.writeFile(wb, filename+'.xlsx');
  showToast('已匯出 '+filename+'.xlsx','ok');
}

function exportXlsxMultisheet(sheetsObj, filename) {
  const wb = XLSX.utils.book_new();
  Object.entries(sheetsObj).forEach(([name, rows]) => {
    if(!rows.length) return;
    const ws = XLSX.utils.json_to_sheet(rows);
    styleXlsxHeader(ws, Object.keys(rows[0]).length);
    XLSX.utils.book_append_sheet(wb, ws, name.replace(/[\\\/\?\*\[\]]/g,''));
  });
  if(wb.SheetNames.length===0) { showToast('無資料可匯出','fail'); return; }
  XLSX.writeFile(wb, filename+'.xlsx');
  showToast('已匯出 '+filename+'.xlsx','ok');
}

function styleXlsxHeader(ws, colCount) {
  ws['!cols'] = Array(colCount).fill({wch:14});
}

async function pushLedgerToGAS() {

  const flatLedger = [];
  Object.keys(ledger).forEach(k=>(ledger[k]||[]).forEach(r=>flatLedger.push({...r,ledgerDate:k})));
  const r = await fetch(gasUrl,{method:'POST',mode:'cors',cache:'no-cache',credentials:'omit',
    body:JSON.stringify({action:'saveLedger',data:flatLedger})});
  if(!r.ok) throw new Error(`HTTP ${r.status}`);
  const d = await r.json();
  if(d.error) throw new Error(d.error);
  return d;
}

async function loadLedgerFromGAS(silent=false) {
  if(!gasUrl) { if(!silent) showToast('請先設定GAS URL','fail'); return; }

    if(!r.ok) throw new Error(`HTTP ${r.status}`);
    const d = await r.json();
    if(d.error) throw new Error(d.error);
    if(d.data) {
      const grouped={};
      d.data.forEach(r=>{
        const key=r.ledgerDate||new Date().toISOString().split('T')[0];
        if(!grouped[key]) grouped[key]=[];

  } catch(e) { if(!silent) showToast('載入失敗: '+e.message,'fail'); }

}

function parseCSVRow(text) {
  const values=[]; let current='',inQuotes=false;
  for(let i=0;i<text.length;i++){
    const ch=text[i];
    if(ch==='"'){if(inQuotes&&text[i+1]==='"'){current+='"';i++;}else{inQuotes=!inQuotes;}}
    else if(ch===','&&!inQuotes){values.push(current);current='';}
    else{current+=ch;}
  }
  values.push(current);
  return values.map(v=>v.replace(/^"|"$/g,'').trim());
}

async function parseUploadedFile(file) {
  const name=file.name.toLowerCase();
  if(name.endsWith('.json')) return JSON.parse(await file.text());
  if(name.endsWith('.csv')) {
    const text=await file.text();
    const lines=text.split(/\r?\n/).filter(l=>l.trim()!=='');
    if(!lines.length) return [];
    const headers=parseCSVRow(lines[0]);
    return lines.slice(1).map(line=>{
      const values=parseCSVRow(line); const row={};
      headers.forEach((h,i)=>{row[h]=values[i]||'';});
      return row;
    });
  }
  if(name.endsWith('.xls')||name.endsWith('.xlsx')) {
    const data=await file.arrayBuffer();
    const wb=XLSX.read(data,{type:'array'});
    return XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]],{defval:''});
  }
  throw new Error('不支援的檔案格式');
}

function normalizeUploadRows(rows) {
  return rows.map(row=>{
    const n={};
    Object.keys(row).forEach(k=>{n[k.trim().replace(/\s+/g,'').toLowerCase()]=row[k];});
    const code=n['代號']||n['code']||n['ticker'];
    const name=n['名稱']||n['name'];
    if(!code||!name) return null;
    return {'代號':String(code),'名稱':String(name),'分類':n['分類']||n['category']||'long_term',
      '現價':parseFloat(n['現價']||n['price']||0)||0,'漲幅':n['漲幅']||n['change']||'',
      '貝氏評分':parseFloat(n['貝氏評分']||n['score']||0)||0};
  }).filter(r=>r&&r['代號']&&r['名稱']);
}

function doGet(e) {
  const a = e.parameter.action;
  if (a === 'ping') return json({status:'ok', time: new Date().toISOString()});
  if (a === 'getLedger') return json({data: readLedger()});
  if (a === 'getPrice') return json({price: getLivePrice(e.parameter.code)});
  if (a === 'getPrices') return json({data: readPrices(e.parameter.date)});
  return json({error:'unknown action'});
}

function doPost(e) {
  try {
    const b = JSON.parse(e.postData.contents);
    if (b.action === 'saveLedger') { writeLedger(b.data); return json({status:'ok', count: b.data.length}); }
    if (b.action === 'savePrices') { writePrices(b.data, b.date); return json({status:'ok'}); }
    return json({error:'unknown action'});
  } catch(err) {
    return json({error: err.message, status: 'error'});
  }
}

// 查詢單檔股票現價（通過台灣證券交易所 API）
function getLivePrice(code) {
  try {
    const tseUrl = \`https://mis.twse.com.tw/stock/api/getStockInfo.jsp?json=1&delay=0&ex_ch=tse_\${code}.tw\`;
    const res = UrlFetchApp.fetch(tseUrl, {muteHttpExceptions: true});
    if (res.getResponseCode() === 200) {
      const data = JSON.parse(res.getContentText());
      const z = data.msgArray?.[0]?.z;
      if (z && z !== '-') return parseFloat(z);
    }
  } catch(e) {}
  
  // 如果 TSE 失敗，嘗試 OTC
  try {
    const otcUrl = \`https://mis.twse.com.tw/stock/api/getStockInfo.jsp?json=1&delay=0&ex_ch=otc_\${code}.tw\`;
    const res = UrlFetchApp.fetch(otcUrl, {muteHttpExceptions: true});
    if (res.getResponseCode() === 200) {
      const data = JSON.parse(res.getContentText());
      const z = data.msgArray?.[0]?.z;
      if (z && z !== '-') return parseFloat(z);
    }
  } catch(e) {}
  
  return null;
}

function getSheet(name, headers) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let s = ss.getSheetByName(name);
  if (!s) { s = ss.insertSheet(name); s.appendRow(headers); s.getRange(1,1,1,headers.length).setFontWeight('bold'); }
  return s;
}

function readLedger() {
  const s = getSheet(LEDGER_SHEET, ['ledgerDate','日期','代號','名稱','分類','買入價','股數','貝氏評分']);
  const rows = s.getDataRange().getValues();
  if (rows.length <= 1) return [];
  return rows.slice(1).map(r => ({
    ledgerDate: String(r[0]), date: fmtDate(r[1]), code: String(r[2]), name: r[3], cat: r[4],
    buyPrice: Number(r[5]), shares: Number(r[6]), score: Number(r[7])
  }));
}

function writeLedger(data) {
  const s = getSheet(LEDGER_SHEET, ['ledgerDate','日期','代號','名稱','分類','買入價','股數','貝氏評分']);
  s.clearContents();
  s.appendRow(['ledgerDate','日期','代號','名稱','分類','買入價','股數','貝氏評分']);
  data.forEach(r => s.appendRow([r.ledgerDate||r.date, r.date, r.code, r.name, r.cat, r.buyPrice, r.shares, r.score]));
}

function readPrices(date) {
  const s = getSheet(PRICES_SHEET, ['日期','代號','名稱','分類','限價','貝氏評分']);
  const rows = s.getDataRange().getValues();
  if (rows.length <= 1) return [];
  return rows.slice(1)
    .filter(r => !date || fmtDate(r[0]) === date)
    .map(r => ({date: fmtDate(r[0]), code: String(r[1]), name: r[2], cat: r[3], price: Number(r[4]), score: Number(r[5])}));
}

function writePrices(data, date) {
  const s = getSheet(PRICES_SHEET, ['日期','代號','名稱','分類','限價','貝氏評分']);
  const rows = s.getDataRange().getValues();
  const toDel = [];
  rows.forEach((r, i) => { if (i > 0 && fmtDate(r[0]) === date) toDel.push(i + 1); });
  toDel.reverse().forEach(i => s.deleteRow(i));
  data.forEach(r => s.appendRow([r.date, r.code, r.name, r.cat, r.price, r.score]));
}

function fmtDate(v) {
  if (v instanceof Date) return Utilities.formatDate(v, 'Asia/Taipei', 'yyyy-MM-dd');
  return String(v);
}`;
}
